{{/* Load site-specific CSS after the theme head without overriding it. - Includes all files under assets/css/custom/*.css - Files are sorted by filename; to control order, prefix with numbers (e.g., 00-base.css, 10-components.css) - Keeps Blowfish head fully intact and upgradable */}} {{- $algo := site.Params.fingerprintAlgorithm | default "sha512" -}} {{- $extras := resources.Match "css/custom/*.css" -}} {{- if $extras -}} {{- $extras = sort $extras "Name" -}} {{- $bundle := $extras | resources.Concat "css/custom.bundle.css" | resources.Minify | resources.Fingerprint $algo -}} {{- end -}}
